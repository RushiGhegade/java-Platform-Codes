// Optimze approch
class MinStack {
    
    Stack<Integer> s;
    Stack<Integer> mins ;
    
    public MinStack() {
        s=new Stack<>();
        mins=new Stack<>();
    }
    
    public void push(int val) {
       if(mins.empty()){
           mins.push(val);
       }else{
           if((int)mins.peek()>=val){
               mins.push(val);
           }   
       }
        
        s.push(val);
    }
    
    public void pop() {
        if((int)mins.peek()==(int)s.peek()){
            mins.pop();
        }
        s.pop();
    }
    
    public int top() {
        return (int)s.peek();
    }
    
    public int getMin() {
        
        if(mins.empty()){
            return -1;
        }
       return (int)mins.peek();
        
    }
}
